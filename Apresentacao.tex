\subsection{Problemas de Planejamento e RTS}
\begin{frame}{Planejamento de Produção (I)}
  \begin{itemize}
    \item StarCraft é dinâmico.
    \item Dinamicidade implica em previsões a partir de fotografias.
    \item Como reproduzir o ambiente dinâmico sem quebrar suas mecânicas?
  \end{itemize}
\end{frame}

\begin{frame}{Planejamento de Produção (II)}
  \begin{itemize}
    \item Problemas de planejamento de projeto restringem os recursos.
    \item Modelos já foram feitos em PDDL \cite{kovarsky06} e usando planejadores \cite{naves12}.
    \item Abordagem genética foi implementada para StarCraft 2, mas uma única raça \cite{kuchem13}.
    \item Modelo mais utilizado ainda é dependente de planejadores e não possui formulação matemática \cite{chan07a}.
  \end{itemize}
\end{frame}

\subsection{Visão Geral}
\begin{frame}{Motivação}
  \begin{itemize}
    \item Poucos modelos multiobjetivos para ambientes similares \cite{captain14}, mas nenhum utiliza recursos como funções-objetivo.
    \item O modelo mais utilizado não possui formulação matemática.
    \item Os modelos utilizados não englobam as características da raça Zerg.
  \end{itemize}
\end{frame}

\begin{frame}{Objetivo}
  Unificar problemas de planejamento de projeto com o modelo de produção de recursos em um problema multiobjetivo que possibilite uma modelagem clara da raça Zerg e utilize recursos como função-objetivo.
\end{frame}

\begin{frame}{Metodologia}
  \begin{itemize}
    \item Implementação de quatro otimizadores: dois algoritmos genéticos, MOACO e MOGRASP.
    \item Dois indicadores foram utilizados para determinar seu desempenho: tempo e hipervolume.
    \item Os parâmetros de cada otimizador foram determinados pelo \emph{irace}.
    \item Os casos de teste foram baseados na raça Zerg.
  \end{itemize}
\end{frame}

\section{Problemas de Planejamento}
\begin{frame}{Problemas de Planejamento}
  Os problemas de planejamento tratados possuem as seguintes características comuns
  \begin{itemize}
    \item O objetivo é minimizar tempo em que a última tarefa termina (\emph{makespan})
    \item Soluções são sequências de tarefas (ou ações).
    \item Existe uma quantidade inicial de recursos que é manipulada por estas tarefas.
  \end{itemize}
\end{frame}

\subsection{Planejamento de Produção online para Produção de Recursos em RTS}
\begin{frame}{Problema de Planejamento de Produção \emph{online} para Produção de Recursos em RTS}
\begin{itemize}
    \item Trata com um conjunto de recursos $R$ e conjunto de restrições $O$.
    \item O jogo muda de estado ao longo do tempo. Cada estado possui quantidade de recursos e lista de ações não-concluídas.
    \item Cada ação manipula recursos através de quatro categorias:
    \begin{itemize}
      \item \emph{Require}
      \item \emph{Burrow}
      \item \emph{Consume}
      \item \emph{Produce}
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Aplicações em StarCraft}
\begin{itemize}
  \item Otimizador baseado no algoritmo SLA* \cite{branq10}.
  \item Algoritmo \emph{branch and bound} \cite{churchill11}.
  \item Novo modelo multiobjetivo \cite{captain14}.
\end{itemize}
\end{frame}

\subsection{Planejamento de Projeto com Restrição de Recursos}
\begin{frame}{Problema de Planejamento de Projeto com Restrição de Recursos}
  \begin{itemize}
    \item Trata com recursos renováveis e não-renováveis.
    \item Possui limite rígido de recursos por todo o horizonte de projeto.
    \item O conjunto de ações do projeto deve ser feito em sua completude.
  \end{itemize}
\end{frame}

\subsection{Planejamento de Produção de Recursos}
\begin{frame}{Problema de Planejamento de Produção de Recursos}
\begin{itemize}
  \item Definido por três variáveis: sistema $S$, estado inicial $I_0$ e estratégia $O$.
  \item Sistema possui o conjunto de recursos $R$ e o conjunto de tarefas $T$.
  \item Estado inicial possui os recursos iniciais e as tarefas não-concluídas.
  \item Estratégia possui as funções-objetivo e as restrições específicas do caso.
\end{itemize}
\end{frame}

\begin{frame}{Recursos}
\begin{itemize}
  \item Cada recurso possui três atributos:
    \begin{itemize}
      \item Máximo coletável, $MC_r$
      \item Máximo utilizável por recurso, $MU_{rs}$
      \item Equivalência, $E_{sr}$
    \end{itemize}
  \item Além de valores referentes a cada atributo, outros valores devem ser guardados para guiar sua manipulação:
    \begin{itemize}
      \item Total coletado do recurso, $C_{rt}$
      \item Total consumido do recurso, $c_{rt}$
      \item Quantidade utilizável do recurso, $U_{rt}$
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Restrições de Recursos}
\begin{align}
  &\bar{C}_{rt} = min\{C_{rt}, MC_r\}\\
  &U_{rt} = \bar{C}_{rt} - u_{rt}
\end{align}

\begin{equation}
    U_{rt} \leq \sum_{s\in R}^{s \neq r} MU_{rs} U_{st}
\end{equation}
\end{frame}

\begin{frame}{Tarefas}
Tarefas possuem os seguintes atributos:
\begin{itemize}
  \item Duração, $d_i$
  \item Pré-requisitos, $Pr_{ir}$
  \item Custo, $Ct_{ir}$
  \item Consumo, $Cm_{ir}$
  \item Uso, $Bo_{ir}$
  \item Produção, $P_{ir}$
\end{itemize}
\end{frame}

\begin{frame}{Restrições de Tarefas}
\begin{align}
  &max\{ U_{rt}, E_{sr}U_{st} | s \in R \} \geq Pr_{ir}\\
  &U_{rt} - B_{rt} \geq Ct_{ir} + Cm_{ir} + Bo_{ir}
\end{align}
\end{frame}

\begin{frame}{Modelo Simplificado}
\begin{align}
  &\text{minimizar } min\left\{C_{max}, UB\right\}\\
  &\text{maximizar } y_m, \forall m \in M\\
  &\text{minimizar } y_n, \forall n \in N, N \cap M = \emptyset
\end{align}
sujeito a
\begin{flalign}
  &S_{max} \leq UB\\
  &y_u > k_u, \forall u \in U\\
  &y_v < l_v, \forall v \in V\\
  &|M| + |N| + |U| + |V| \geq 1\\
  &UB > 0
\end{flalign}
\end{frame}

\begin{frame}{Restrições Gerais}
\footnotesize{
\begin{flalign}
  &C_{rt} = C_{r0} + \sum^{i \in x}_i \sum^{t}_{\tau=1} x_{i(\tau-d_i)} P_{ir}, \forall t \in \{1, \dots, UB\}, \forall r \in R\\
  &u_{rt} = u_{r0} + \sum^{i \in x}_i \sum^{t}_{\tau=1}
    \left(
      x_{i(\tau-d_i)} Cm_{ir} +
      x_{i\tau} Ct_{ir}
    \right), \forall t \in \{1, \dots, UB\}, \forall r \in R\\
  &B_{rt} = \sum^{i \in x}_i \sum^{t}_{\tau=1}
  \left(
    x_{i\tau} - x_{i(\tau-d_i)}
  \right)
  \left(
    Bo_{ir} + Cm_{ir}
  \right),
  \forall t \in \{1, \dots, UB\}, \forall r \in R\\
  &U_{rt} \leq \sum_{s\in R}^{s \neq r} MU_{rs} U_{st}, \forall t \in \{1, \dots, UB\}, \forall r \in R\\
  &max\{ U_{rt}, E_{sr}U_{st} | s \in R \} \geq \sum^{i \in x}_i x_{it}Pr_{ir}, \forall t \in \{1, \dots, UB\}, \forall r \in R\\
  &U_{rt} - B_{rt} \geq \sum^{i \in x}_i x_{it} (Ct_{ir} + Cm_{ir} + Bo_{ir}), \forall t \in \{1, \dots, UB\}, \forall r \in R
\end{flalign}}
\end{frame}

\subsection{Planejamento de Produção de Recursos com Eventos Cíclicos}
\begin{frame}{Problema de Planejamento de Produção de Recursos com Eventos Cíclicos}
\begin{itemize}
  \item O problema de planejamento de produção de recursos ainda não reproduz as condições da raça Zerg.
  \item Tarefas que se repetem crescem o tamanho da solução de muito mais rápido que tarefas relevantes.
  \item Para resolver os dois problemas: eventos cíclicos.
\end{itemize}
\end{frame}

\begin{frame}{Eventos Cíclicos}
Eventos possuem os seguintes atributos:
\begin{itemize}
  \item Tempo de espera (\emph{cooldown}), $d^e_i$
  \item Produção, $P^e_{ir}$
  \item Gatilho, $g^e_i$
\end{itemize}
\end{frame}

\begin{frame}{Alterações nas Restrições}
\begin{align}
  &C_{rt} = C_{r0} +
  \sum^{i \in x}_i \sum^{t}_{\tau=1} x_{i(\tau-d_i)} P_{ir} + P^e_{rt},\forall t \in \{1, \dots, UB\}, \forall r \in R\\
  &\sum^t_{\tau=1} e_{i\tau} = min\{C_{g^e_i t}, MC_{g^e_i}\}, \forall t \in \{1, \dots, UB\}, \forall i \in x^e\\
  &\sum^t_{\tau=1} \bar{e}_{i\tau} = u_{g^e_i t} + B_{g^e_it}, \forall t \in \{1, \dots, UB\}, \forall i \in x^e
\end{align}
\end{frame}

\subsection{Valoração das Tarefas}
\begin{frame}{Pesos}
\begin{itemize}
  \item $\delta_o$: objetivos
  \item $\delta_r$: restrições
  \item $\delta_p$: pré-requisitos e usos
  \item $\delta_c$: custos e consumos
  \item $\delta_m$: aumento de máximo utilizável de um recurso
\end{itemize}
\end{frame}

\begin{frame}{Valores Iniciais}
Cada tarefa recebe:
\begin{itemize}
  \item $\delta_o$ para cada tipo de recurso cujo objetivo seja ajudado
  \item $-\frac{\delta_o}{2}$ para cada recurso cujo objetivo seja atrapalhado
  \item $\delta_r$ para cada recurso que ajude a cumprir restrições
  \item $-\frac{\delta_r}{2}$ para cada recurso cujas restrições sejam atrapalhadas
\end{itemize}
\end{frame}

\begin{frame}{Algoritmo de Valoração}
\begin{center}
  \scalebox{.65}{
    \begin{minipage}{1\linewidth}
      \begin{algorithm}[H]
      \begin{algorithmic}[1]
      \Procedure{getValues}
      {$I_0$, $O$, $\delta_o$, $\delta_r$, $\delta_p$, $\delta_c$, $\delta_m$}

      \State $initial \gets$ \Call{initialWeights}
        {$\delta_o$, $\delta_r$, $I_0$, $O$}
      \State $taskValues \gets initial$
      \State $graph \gets$ \Call{initialGraph}{$S$}
      \State \Call{pruneGraph}{$prerequisites$}
      \State \Call{pruneGraph}{$borrow$}
      \State \Call{getWeights}{$I_0$, $\delta_p$, $\delta_c$, $\delta_m$}
      \For{$i = 1$ \textbf{to} $T$}
        \If{$initial[i] > 0$}
          \State \Call{passValues}
            {$i$, $\delta_p$, $\delta_c$, $\delta_m$, $taskValues$}
        \EndIf
      \EndFor
      \For{$i = 1$ \textbf{to} $T$}
        \If{\Call{not\_possible}{$I_0$, $i$}}
          \State $initial[i] \gets -\infty$
        \EndIf
      \EndFor
      \State $difference \gets$ \Call{greatestNegative}{$taskValues$}
      \State \Call{add}{$taskValues$, $difference$}
      \State \Call{normalize}{$taskValues$, $x > 0$}
      \State \Return $taskValues$
      \EndProcedure
      \end{algorithmic}
      %\caption{Algoritmo de valoração de tarefas em um estado $I_0$ segundo uma estratégia $O$.}
      \end{algorithm}
    \end{minipage}
  }
\end{center}
\end{frame}

\begin{frame}{Pesos das Arestas}
\begin{equation}
w_{e} = \delta
  \left[
    \sigma +
    \left(
      tanh\left(\frac{B_e}{C_e}\right)
      \frac{B_e}{B_e + A_e}
    \right)^2
  \right]
\end{equation}

Onde
\begin{itemize}
  \item $\delta$ é o peso correspondente ao tipo de aresta,
  \item $\sigma$ recebe $0,3$ se houver evento que produza recurso que liga as tarefas,
  \item $B_e$ é o bônus associado à tarefa-alvo,
  \item $A_e$ é a base de comparação para $B_e$,
  \item $C_e$ é a necessidade da tarefa origem.
\end{itemize}
\end{frame}

\subsection{Algoritmo de Criação}
\begin{frame}{Algoritmo de Criação}
\begin{center}
  \scalebox{.65}{
    \begin{minipage}{1\linewidth}
\begin{algorithm}[H]
\begin{algorithmic}[1]
\Procedure{create}
{$I_0$, $O$, $\delta_o$, $\delta_r$, $\delta_p$, $\delta_c$, $\delta_m$, $\Delta_o$, $\Delta_r$}
\State $x \gets \{\}$
\State $r \gets 1$
\While{\Call{not\_valid}{$x$, $O$}}
  \State $values \gets$ \Call{getValues}
    {$I_0$, $O$, $\delta_o$, $\delta_r$, $\delta_p$, $\delta_c$, $\delta_m$}
  \State $\delta_o \gets \delta_o \Delta_o$
  \State $\delta_r \gets \delta_r \Delta_r$
  \State $x \gets x$ $\cup$ \Call{roulette}{values}
  \If{$S_{max}(x) > UB$}
    \State $size \gets$ \Call{$x.$size}{}
    \State \Call{$x$.resize}{$size - r$}
    \State $r \gets r + 1$
    \If{$r >$ \Call{$x.$size}{}}
      \State $r \gets 1$
    \EndIf
  \EndIf
\EndWhile
\State \Return $x$
\EndProcedure
\end{algorithmic}
\end{algorithm}
    \end{minipage}
  }
\end{center}
\end{frame}

\section{Otimizadores}
\begin{frame}{Otimizadores}
Foram implementados quatro otimizadores:
\begin{itemize}
    \item NSGA-II -- Non-Dominated Sorting Genetic Algorithm
    \item ``Joelho'' -- Variação do NSGA-II
  \item MOACO -- Multi-Objective Ant Colony Optimization
  \item MOGRASP -- Multi-Objective Greedy Random Adaptive Search Procedure
\end{itemize}
\end{frame}

\subsection{Algoritmos Genéticos}
\begin{frame}{Características dos Algoritmos Genéticos}
\begin{itemize}
  \item Não possui \emph{crossover}
  \item Mutação é obrigatória, ao invés de probabilística
  \item Implementação paralela
\end{itemize}
\end{frame}

\begin{frame}{Algoritmo Genético Multiobjetivo}
\begin{center}
  \scalebox{.65}{
    \begin{minipage}{1\linewidth}
\begin{algorithm}[H]
\begin{algorithmic}
\Procedure{MOGA}{$I_0$, $O$, $n$, $\delta_o^F$, $\delta_r^F$, $\Delta_o^F$, $\Delta_r^F$}
\State $x \gets$ \Call{generateParallel}{$n$}
\State \Call{evaluateAndSort}{$x$}
\For{$i = 1$ \textbf{to} $it$}
  \State $x \gets$ \Call{select}{$x$}
  \State $\dot{x} \gets \emptyset$
  \ForAll{$s \in x$ \emph{in parallel}}
    \State $\dot{x} \gets \dot{x}$ $\cup$ \Call{mutate}{$s$, $\delta_o^F$, $\delta_r^F$, $\Delta_o^F$, $\Delta_r^F$}
  \EndFor
  \State $x \gets x$ $\cup$ $\dot{x}$
  \State \Call{evaluateAndSort}{$x$}
\EndFor
\EndProcedure
\end{algorithmic}
\label{alg:nsga}
\end{algorithm}
    \end{minipage}
  }
\end{center}
\end{frame}

\begin{frame}{Variações}
Duas variações foram implementadas, diferindo no algoritmo de seleção de soluções.
  \begin{itemize}
    \item NSGA-II: utiliza o ordenamento de não-dominância.
    \item ``Joelho'': utiliza o ângulo entre soluções como parâmetro de ordenação.
  \end{itemize}
\end{frame}

\begin{algorithm}[H]
\begin{algorithmic}
\Procedure{MOACO}{$I_0$, $O$, $\delta_o$, $\delta_r$, $\Delta_o$, $\Delta_r$, $\rho$, $\alpha$, $\beta$, $n$, $\delta_o^F$, $\delta_r^F$, $\Delta_o^F$, $\Delta_r^F$}

\EndFor

\EndProcedure

\section{Conclusão}
\begin{frame}{Conclusão}
\begin{itemize}
  \item Um novo problema de otimização multiobjetivo foi mostrado.
  \item Este modelo consegue simular situações reais de um jogo RTS.
  \item Quatro otimizadores foram implementados para o problema.
  \item NSGA-II apresentou o melhor desempenho em tempo e hipervolumes.
  \item Trabalhos futuros incluem:
  \begin{itemize}
    \item Criação de tomador de decisão;
    \item Implementação em \emph{game AI} para StarCraft: Brood War;
    \item Operadores mais complexos para algoritmos genéticos.
  \end{itemize}
\end{itemize}
\end{frame}
